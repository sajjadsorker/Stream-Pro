// --- UTILITY: M3U PARSER ---
const parseM3U = (m3uString) => {
  const lines = m3uString.split('\n');
  const channels = [];
  let currentChannel = null;

  lines.forEach(line => {
    line = line.trim();
    if (line.startsWith('#EXTINF:')) {
      const tvgLogoMatch = line.match(/tvg-logo="([^"]+)"/);
      const groupTitleMatch = line.match(/group-title="([^"]+)"/);
      const nameMatch = line.split(',').pop();

      currentChannel = {
        id: crypto.randomUUID(),
        name: nameMatch ? nameMatch.trim() : 'Unknown Channel',
        logo: tvgLogoMatch ? tvgLogoMatch[1] : 'https://via.placeholder.com/150?text=TV',
        category: groupTitleMatch ? groupTitleMatch[1] : 'Imported',
        language: 'Unknown',
        country: 'Unknown',
        streamUrl: '',
        favorite: false
      };
    } else if (line.startsWith('http') && currentChannel) {
      currentChannel.streamUrl = line;
      // Ensure category exists or default to Imported
      if (!CATEGORIES.includes(currentChannel.category)) {
          currentChannel.category = 'Imported';
      }
      channels.push({ ...currentChannel });
      currentChannel = null;
    }
  });
  return channels;
};

// 1. HLS Video Player Component
const HlsPlayer = ({ channel, onPlay }) => {
  const videoRef = useRef(null);
  const [isPlaying, setIsPlaying] = useState(true);
  const [isMuted, setIsMuted] = useState(false);
  const [error, setError] = useState(false);

  useEffect(() => {
    if (!channel?.streamUrl) return;
    setError(false);
    
    const video = videoRef.current;
    let hls;

    const initPlayer = () => {
      if (video.canPlayType('application/vnd.apple.mpegurl')) {
        // Safari native HLS support
        video.src = channel.streamUrl;
        video.play().catch(e => console.log("Autoplay prevented"));
      } else {
        // Use HLS.js for Chrome/Firefox/Edge
        if (!window.Hls) {
           const script = document.createElement('script');
           script.src = 'https://cdn.jsdelivr.net/npm/hls.js@latest';
           script.onload = () => setupHls();
           document.body.appendChild(script);
        } else {
           setupHls();
        }
      }
    };

    const setupHls = () => {
      if (window.Hls.isSupported()) {
        hls = new window.Hls({ enableWorker: true });
        hls.loadSource(channel.streamUrl);
        hls.attachMedia(video);
        hls.on(window.Hls.Events.MANIFEST_PARSED, () => {
          video.play().catch(e => console.log("Autoplay prevented", e));
        });
        hls.on(window.Hls.Events.ERROR, (event, data) => {
          if (data.fatal) setError(true);
        });
      }
    };

    initPlayer();
    if(onPlay) onPlay(channel);

    return () => {
      if (hls) hls.destroy();
    };
  }, [channel]);

  const togglePlay = () => {
    if (videoRef.current.paused) { videoRef.current.play(); setIsPlaying(true); } 
    else { videoRef.current.pause(); setIsPlaying(false); }
  };

  const toggleMute = () => {
    videoRef.current.muted = !isMuted;
    setIsMuted(!isMuted);
  };

  const toggleFullScreen = () => {
    if (videoRef.current.requestFullscreen) {
      videoRef.current.requestFullscreen();
    }
  };

  if (!channel) return (
    <div className="w-full h-full bg-[#050810] flex items-center justify-center rounded-2xl border border-white/5 shadow-2xl">
      <div className="text-center text-white/40 flex flex-col items-center gap-4">
        <Tv size={48} className="opacity-50" />
        <p className="text-lg font-medium">Select a channel to start streaming</p>
      </div>
    </div>
  );

  return (
    <div className="relative w-full h-full bg-black rounded-2xl overflow-hidden shadow-[0_0_40px_rgba(212,175,55,0.05)] group border border-white/10">
      {error && (
        <div className="absolute inset-0 z-20 flex items-center justify-center bg-black/80 backdrop-blur-sm">
           <div className="text-center text-red-400">
              <AlertCircle size={48} className="mx-auto mb-3" />
              <p>Stream offline or unavailable.</p>
           </div>
        </div>
      )}
      
      <video 
        ref={videoRef} 
        className="w-full h-full object-contain bg-black"
        onPlay={() => setIsPlaying(true)}
        onPause={() => setIsPlaying(false)}
      />
      
      {/* Overlays */}
      <div className="absolute top-4 left-4 z-10 flex gap-3 items-center">
        <div className="bg-red-600 text-white text-xs font-bold px-3 py-1 rounded-full animate-pulse flex items-center gap-2 shadow-lg">
          <div className="w-2 h-2 bg-white rounded-full"></div> LIVE
        </div>
        <div className="bg-black/60 backdrop-blur-md px-4 py-1.5 rounded-full border border-white/10 text-white flex items-center gap-2">
          <img src={channel.logo} alt={channel.name} className="h-5 w-5 object-contain" />
          <span className="font-medium text-sm">{channel.name}</span>
        </div>
      </div>

      {/* Custom Controls (Hover) */}
      <div className="absolute bottom-0 left-0 right-0 bg-gradient-to-t from-black/90 via-black/40 to-transparent p-6 opacity-0 group-hover:opacity-100 transition-opacity duration-300 z-10 flex items-end justify-between">
        <div className="flex gap-4 items-center text-white">
           <button onClick={togglePlay} className="hover:text-amber-400 transition transform hover:scale-110">
             {isPlaying ? <Pause size={28} /> : <Play size={28} />}
           </button>
           <button onClick={toggleMute} className="hover:text-amber-400 transition transform hover:scale-110">
             {isMuted ? <VolumeX size={24} /> : <Volume2 size={24} />}
           </button>
           <div className="text-sm font-medium opacity-80">{channel.category}</div>
        </div>
        <button onClick={toggleFullScreen} className="text-white hover:text-amber-400 transition transform hover:scale-110">
          <Maximize size={24} />
        </button>
      </div>
    </div>
  );
};

                <div className="flex items-center justify-between mb-6">
                  <h2 className="text-2xl font-bold text-white">
                    {activeCategory === 'Favorites' ? 'My Favorites' :

                  <span className="text-sm text-gray-400 bg-white/5 px-3 py-1 rounded-full border border-white/5">

                        <Heart size={16} className={favoriteIds.includes(channel.id) ? "fill-amber-500 text-amber-500" : "text-white"} />

                        <div className="absolute inset-0 border-2 border-amber-500 rounded-2xl z-10 pointer-events-none"></div>

                        <h3 className="font-semibold text-white truncate text-base group-hover:text-amber-400 transition-colors">{channel.name}</h3>

                          <span className="bg-white/5 px-2 py-0.5 rounded-md">{channel.category}</span>
                          <span className="uppercase tracking-wider">{channel.language}</span>
                        </div>
                      </div>
                    </div>
                  ))}
                  
                  {filteredChannels.length === 0 && (
                    <div className="col-span-full py-16 text-center text-gray-500 bg-white/5 border border-white/5 rounded-2xl border-dashed">
                       <ListVideo size={48} className="mx-auto mb-4 opacity-50" />
                       <h3 className="text-lg font-medium text-white mb-2">কোন চ্যানেল পাওয়া যায়নি (No channels found)</h3>
                       <p>সার্চ বা ক্যাটাগরি পরিবর্তন করে দেখুন।</p>
                    </div>
                  )}
                </div>
              </section>

              {/* Footer Ad */}
              <AdBanner title="Footer Banner" size="728x90" code={adCodes.footer} />

              {/* Popunder Ad (Invisible container) */}
              {adCodes.popunder && (
                 <div dangerouslySetInnerHTML={{ __html: adCodes.popunder }} />
              )}
            </>
          ) : (
            
            /* ---------------- ADMIN DASHBOARD ---------------- */
            <div className="max-w-6xl mx-auto space-y-8 pb-10">
              <div className="flex items-center justify-between">
                 <div>
                   <h1 className="text-3xl font-bold text-white flex items-center gap-3">
                     <Settings className="text-amber-500" /> Admin Control Panel
                   </h1>
                   <p className="text-gray-400 mt-2">Manage channels, bulk import M3U playlists, and monitor system.</p>
                 </div>
                 <div className="bg-amber-500/10 border border-amber-500/20 text-amber-500 px-4 py-2 rounded-xl text-sm font-semibold flex items-center gap-2">
                    <Shield size={16} /> Authenticated
                 </div>
              </div>

              <div className="grid lg:grid-cols-3 gap-6">
                 {/* M3U Import Tool */}
                 <div className="bg-[#111827] border border-white/5 rounded-2xl p-6 lg:col-span-1 shadow-xl">
                   <h2 className="text-xl font-semibold text-white mb-4 flex items-center gap-2">
                     <Upload size={20} className="text-blue-400" /> M3U Importer
                   </h2>
                   <form onSubmit={handleM3uImport}>
                      <textarea 
                        name="m3uText"
                        placeholder="#EXTM3U&#10;#EXTINF:-1 tvg-logo=&quot;url&quot; group-title=&quot;News&quot;,Channel Name&#10;http://stream.m3u8"

                        required
                      ></textarea>
                      <button type="submit" className="w-full bg-blue-600 hover:bg-blue-700 text-white font-medium py-3 rounded-xl transition-colors flex items-center justify-center gap-2 shadow-lg shadow-blue-900/20">
                         Parse & Import Playlist
                      </button>
                   </form>
                   <div className="mt-4 text-xs text-gray-500 bg-blue-500/5 p-3 rounded-lg border border-blue-500/10">
                     <strong className="text-blue-400">Pro Tip:</strong> Paste raw M3U text here. The system will automatically extract logos, names, categories, and stream URLs.
                   </div>
                 </div>

                 {/* Channel Management */}
                 <div className="bg-[#111827] border border-white/5 rounded-2xl p-0 lg:col-span-2 shadow-xl overflow-hidden flex flex-col">
                   <div className="p-6 border-b border-white/5 flex flex-wrap items-center justify-between gap-4 bg-white/[0.02]">
                     <h2 className="text-xl font-semibold text-white flex items-center gap-2">
                       <Tv size={20} className="text-emerald-400" /> Database Management
                     </h2>
                     <div className="flex items-center gap-3">
                       <button onClick={loadDefaultChannels} className="text-xs bg-blue-600 hover:bg-blue-700 text-white px-3 py-1.5 rounded-lg transition-colors">
                          Load Default Channels
                       </button>
                       <span className="text-sm font-medium bg-emerald-500/10 text-emerald-400 px-3 py-1 rounded-full border border-emerald-500/20">
                         {channels.length} Total Channels
                       </span>
                     </div>
                   </div>
                   
                   <div className="flex-1 overflow-auto bg-[#050810]/50 min-h-[400px]">
                     <table className="w-full text-left text-sm text-gray-400">
                       <thead className="text-xs uppercase bg-[#111827] text-gray-500 sticky top-0 z-10 shadow-md">
                         <tr>
                           <th className="px-6 py-4 font-semibold">Logo & Name</th>
                           <th className="px-6 py-4 font-semibold">Category</th>
                           <th className="px-6 py-4 font-semibold">Stream URL</th>
                           <th className="px-6 py-4 font-semibold text-right">Actions</th>
                         </tr>
                       </thead>
                       <tbody className="divide-y divide-white/5">
                         {channels.map((channel, i) => (
                           <tr key={channel.id} className="hover:bg-white/[0.02] transition-colors group">
                             <td className="px-6 py-4 flex items-center gap-4">
                                <img src={channel.logo} alt="" className="w-10 h-10 object-contain bg-black/50 p-1 rounded-lg border border-white/5" />

                             </td>
                             <td className="px-6 py-4">
                               <span className="bg-white/5 px-2.5 py-1 rounded-md text-xs border border-white/5">{channel.category}</span>
                             </td>
                             <td className="px-6 py-4 font-mono text-xs max-w-[200px] truncate text-gray-500">
                               {channel.streamUrl}
                             </td>
                             <td className="px-6 py-4 text-right space-x-2">
                               <button 
                                 onClick={() => handleEditChannel(channel)}
                                 className="p-2 text-blue-400 hover:bg-blue-400/10 rounded-lg transition-colors opacity-0 group-hover:opacity-100"
                               >
                                 <Edit2 size={16} />
                               </button>
                               <button 
                                 onClick={() => handleDeleteChannel(channel.id)}
                                 className="p-2 text-red-400 hover:bg-red-400/10 rounded-lg transition-colors opacity-0 group-hover:opacity-100"
                               >
                                 <Trash2 size={16} />
                               </button>
                             </td>
                           </tr>
                         ))}
                       </tbody>
                     </table>
                   </div>
                 </div>
              </div>

              {/* Ads Manager Section */}
              <div className="mt-8 bg-[#111827] border border-white/5 rounded-2xl p-6 shadow-xl">
                 <h2 className="text-xl font-semibold text-white mb-6 flex items-center gap-2">
                   <DollarSign size={24} className="text-green-400" /> অ্যাড ম্যানেজার (Ads Manager)
                 </h2>
                 <form onSubmit={handleSaveAds} className="space-y-6">
                    <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
                      {/* Header Banner */}
                      <div className="space-y-2">
                         <label className="text-sm font-medium text-gray-400">Header Banner (728×90)</label>
                         <textarea 
                           value={adCodes.header}
                           onChange={(e) => setAdCodes({...adCodes, header: e.target.value})}
                           placeholder="Adsterra script code here..."
                           className="w-full h-24 bg-[#050810] border border-white/10 rounded-xl p-3 text-xs font-mono text-gray-300 focus:outline-none focus:border-green-500 resize-none"
                         ></textarea>
                      </div>

                      {/* Below Player */}
                      <div className="space-y-2">
                         <label className="text-sm font-medium text-gray-400">Below Player (300×250)</label>
                         <textarea 
                           value={adCodes.belowPlayer}
                           onChange={(e) => setAdCodes({...adCodes, belowPlayer: e.target.value})}
                           placeholder="Adsterra script code here..."
                           className="w-full h-24 bg-[#050810] border border-white/10 rounded-xl p-3 text-xs font-mono text-gray-300 focus:outline-none focus:border-green-500 resize-none"
                         ></textarea>
                      </div>

                      {/* Footer Banner */}
                      <div className="space-y-2">
                         <label className="text-sm font-medium text-gray-400">Footer Banner (728×90)</label>
                         <textarea 
                           value={adCodes.footer}
                           onChange={(e) => setAdCodes({...adCodes, footer: e.target.value})}
                           placeholder="Adsterra script code here..."
                           className="w-full h-24 bg-[#050810] border border-white/10 rounded-xl p-3 text-xs font-mono text-gray-300 focus:outline-none focus:border-green-500 resize-none"
                         ></textarea>
                      </div>

                      {/* Popunder / Social Bar */}
                      <div className="space-y-2">
                         <label className="text-sm font-medium text-gray-400">Popunder / Social Bar</label>
                         <textarea 
                           value={adCodes.popunder}
                           onChange={(e) => setAdCodes({...adCodes, popunder: e.target.value})}
                           placeholder="Adsterra popunder script here..."
                           className="w-full h-24 bg-[#050810] border border-white/10 rounded-xl p-3 text-xs font-mono text-gray-300 focus:outline-none focus:border-green-500 resize-none"
                         ></textarea>
                      </div>
                    </div>
                    
                    <button type="submit" className="w-full md:w-auto px-8 bg-green-600 hover:bg-green-700 text-white font-medium py-3 rounded-xl transition-colors flex items-center justify-center gap-2 shadow-lg shadow-green-900/20">
                       <DollarSign size={18} /> অ্যাড কোড সেভ করুন
                    </button>
                 </form>
              </div>
            </div>
          )}
        </div>
      </main>

      {/* Embedded CSS for custom scrollbar to keep it single file without complex tailwind plugins */}
      <style dangerouslySetInnerHTML={{__html: `
        .scrollbar-hide::-webkit-scrollbar {
            display: none;
        }

            -ms-overflow-style: none;

            100% { transform: translateX(100%); }
        }
      `}} />
    </div>
  );
